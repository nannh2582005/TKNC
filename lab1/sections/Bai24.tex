\begin{question}{2.4}

Cho
\[
X=(X_1,X_2,X_3)' \sim N_3(\mu,\Sigma)
\]

với
\[
\mu=
\begin{pmatrix}
2\\
-3\\
1
\end{pmatrix},
\qquad
\Sigma=
\begin{pmatrix}
1&1&1\\
1&3&2\\
1&2&2
\end{pmatrix}.
\]

\begin{enumerate}
\item[(a)] Xác định phân phối của $3X_1-2X_2+X_3$.
\item[(b)] Tìm vector $a\in\mathbb{R}^2$ sao cho $X_2$ và 
$X_2-a'\begin{pmatrix}X_1\\X_3\end{pmatrix}$ độc lập.
\end{enumerate}

\textbf{Lời giải:}

%------2.4a------
\begin{subquestion}
\begin{answer}

Đặt
\[
Y=3X_1-2X_2+X_3=c'X,
\qquad
c=
\begin{pmatrix}
3\\
-2\\
1
\end{pmatrix}.
\]

Vì $X\sim N_3(\mu,\Sigma)$ nên

\[
Y \sim N(c'\mu,c'\Sigma c).
\]

Ta có

\[
c'\mu=3\cdot2-2(-3)+1=13.
\]

\[
\Sigma c=
\begin{pmatrix}
1&1&1\\
1&3&2\\
1&2&2
\end{pmatrix}
\begin{pmatrix}
3\\
-2\\
1
\end{pmatrix}
=
\begin{pmatrix}
2\\
-1\\
1
\end{pmatrix}.
\]

\[
c'\Sigma c=
\begin{pmatrix}
3&-2&1
\end{pmatrix}
\begin{pmatrix}
2\\
-1\\
1
\end{pmatrix}
=9.
\]

Do đó

\[
3X_1-2X_2+X_3 \sim N(13,9).
\]

\end{answer}
\end{subquestion}

%------2.4b------
\begin{subquestion}
\begin{answer}

Đặt
\[
a=
\begin{pmatrix}
a_1\\
a_2
\end{pmatrix}.
\]

Khi đó

\[
Z=X_2-a_1X_1-a_2X_3.
\]

Vì $(X_1,X_2,X_3)$ có phân phối chuẩn nhiều chiều nên  
$X_2$ và $Z$ độc lập khi

\[
Cov(X_2,Z)=0.
\]

Ta có

\[
Cov(X_2,Z)
=Var(X_2)-a_1Cov(X_2,X_1)-a_2Cov(X_2,X_3).
\]

Từ $\Sigma$

\[
Var(X_2)=3,\qquad Cov(X_2,X_1)=1,\qquad Cov(X_2,X_3)=2.
\]

Do đó

\[
3-a_1-2a_2=0
\]

\[
\Leftrightarrow\; a_1+2a_2=3.
\]

Vậy mọi vector

\[
a=
\begin{pmatrix}
a_1\\
a_2
\end{pmatrix}
\in\mathbb{R}^2
\text{ thỏa } a_1+2a_2=3
\]

đều làm cho hai biến độc lập.

\end{answer}
\end{subquestion}

\end{question}

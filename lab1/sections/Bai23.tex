\begin{question}{2.3}
Cho
\[
X = (X_1, X_2, X_3)' \sim N_3(\mu, \Sigma)
\]

với
\[
\mu =
\begin{pmatrix}
1\\
-1\\
2
\end{pmatrix},
\qquad
\Sigma =
\begin{pmatrix}
4 & 0 & -1\\
0 & 5 & 0\\
-1 & 0 & 2
\end{pmatrix}.
\]

Xác định phân phối có điều kiện của:

\begin{enumerate}
\item[(a)] $X_1 \text{ khi biết } X_3 = x_3$
\item[(b)] $X_1 \text{ khi biết } X_2 = x_2 \text{ và } X_3 = x_3$
\item[(c)] $X_2 \text{ khi biết } X_1 = x_1 \text{ và } X_3 = x_3$
\end{enumerate}

\textbf{Lời giải:}

%--------2.3a--------
\begin{subquestion}
\begin{answer}
Ta có
\[
X = (X_1, X_2, X_3)' \sim N_3(\mu,\Sigma).
\]

Phân phối điều kiện của $X_1$ khi biết $X_3=x_3$ trong phân phối chuẩn nhiều chiều là
\[
X_1 \mid X_3 = x_3 \sim \mathcal{N}(\mu_{1|3}, \sigma^2_{1|3})
\]

trong đó
\[
\mu_{1|3} = \mu_1 + \frac{\sigma_{13}}{\sigma_{33}}(x_3 - \mu_3)
\]

\[
\sigma^2_{1|3} = \sigma_{11} - \frac{\sigma_{13}^2}{\sigma_{33}}
\]

Từ ma trận $\Sigma$ ta có
\[
\sigma_{11}=4, \qquad \sigma_{33}=2, \qquad \sigma_{13}=-1.
\]

Suy ra
\[
\mu_{1|3} = 1 + \frac{-1}{2}(x_3-2)
\]

\[
\mu_{1|3} = 2 - \frac{x_3}{2}
\]

và
\[
\sigma^2_{1|3} = 4 - \frac{(-1)^2}{2} = \frac{7}{2}.
\]

Do đó
\[
X_1 \mid X_3 = x_3 \sim \mathcal{N}\left(2-\frac{x_3}{2}, \frac{7}{2}\right).
\]
\end{answer}
\end{subquestion} 

%--------2.3b--------
\begin{subquestion}
\begin{answer}
    Câu 2.3b
\end{answer}
\end{subquestion}

%--------2.3c--------
\begin{subquestion}
\begin{answer}

Vì $X$ có phân phối chuẩn nhiều chiều nên phân phối có điều kiện của một thành phần khi biết các thành phần còn lại vẫn là phân phối chuẩn.

Đặt
\[
Y = X_2, \qquad
Z =
\begin{pmatrix}
X_1 \\
X_3
\end{pmatrix}.
\]

Khi đó, theo công thức phân phối có điều kiện của phân phối chuẩn nhiều chiều, ta có
\[
Y \mid Z = z \sim N\left(
\mu_Y + \Sigma_{YZ}\Sigma_{ZZ}^{-1}(z - \mu_Z),\;
\Sigma_{YY} - \Sigma_{YZ}\Sigma_{ZZ}^{-1}\Sigma_{ZY}
\right).
\]

- Vì
\[
\mu =
\begin{pmatrix}
1\\
-1\\
2
\end{pmatrix},
\]
nên
\[
\mu_Y = E(X_2) = -1, \qquad
\mu_Z =
\begin{pmatrix}
E(X_1)\\
E(X_3)
\end{pmatrix}
=
\begin{pmatrix}
1\\
2
\end{pmatrix}.
\]

- Từ
\[
\Sigma =
\begin{pmatrix}
4 & 0 & -1\\
0 & 5 & 0\\
-1 & 0 & 2
\end{pmatrix},
\]

ta có
\begin{itemize}
    \item $\Sigma_{YY} = \mathrm{Var}(X_2) = 5$.
    \item $\Sigma_{YZ} = 
        \begin{pmatrix} 
        \mathrm{Cov}(X_2, X_1) & \mathrm{Cov}(X_2, X_3)
        \end{pmatrix} = 
        \begin{pmatrix}
        0 & 0
        \end{pmatrix}$.
    \item 
        \[
        \Sigma_{ZZ} =
        \begin{pmatrix}
        \mathrm{Var}(X_1) & \mathrm{Cov}(X_1, X_3)\\
        \mathrm{Cov}(X_3, X_1) & \mathrm{Var}(X_3)
        \end{pmatrix}
        =
        \begin{pmatrix}
        4 & -1\\
        -1 & 2
        \end{pmatrix}.
        \]
\end{itemize}

\textbf{Kỳ vọng có điều kiện:}

Theo công thức,
\[
E(X_2 \mid X_1 = x_1, X_3 = x_3)
= \mu_Y + \Sigma_{YZ}\Sigma_{ZZ}^{-1}
\left[
\begin{pmatrix}
x_1\\
x_3
\end{pmatrix}
-
\begin{pmatrix}
1\\
2
\end{pmatrix}
\right].
\]

Vì
\[
\Sigma_{YZ} = (0 \;\; 0),
\]
nên
\[
\Sigma_{YZ}\Sigma_{ZZ}^{-1} = (0 \;\; 0).
\]

Do đó
\[
E(X_2 \mid X_1 = x_1, X_3 = x_3) = -1.
\]

\textbf{Phương sai có điều kiện:}

Theo công thức,
\[
\mathrm{Var}(X_2 \mid X_1 = x_1, X_3 = x_3)
= \Sigma_{YY} - \Sigma_{YZ}\Sigma_{ZZ}^{-1}\Sigma_{ZY}.
\]

Vì
\[
\Sigma_{YZ} = (0 \;\; 0),
\]
nên
\[
\Sigma_{YZ}\Sigma_{ZZ}^{-1}\Sigma_{ZY} = 0.
\]

Suy ra
\[
\mathrm{Var}(X_2 \mid X_1 = x_1, X_3 = x_3) = 5.
\]

\textbf{Vậy}
\[
X_2 \mid (X_1 = x_1, X_3 = x_3) \sim N(-1, 5).
\]

\end{answer}
\end{subquestion}

\end{question}
